📊 តភ្ជាប់ Google Sheet

រកក្នុង URL: docs.google.com/spreadsheets/d/[SHEET_ID]/edit

Extensions → Apps Script → Deploy → New Deployment → Web App → Anyone

📋 Apps Script Code (ចម្លងទៅ Google Apps Script):
// ── GET: Read data (supports JSONP for CORS-free Pull) ── function doGet(e) { var tab = (e.parameter.tab) ? e.parameter.tab : 'Staff'; var cb = e.parameter.callback; // JSONP callback name var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(tab); if (!sh) sh = ss.insertSheet(tab); var data = sh.getLastRow() > 0 ? sh.getDataRange().getValues() : []; var json = JSON.stringify({status:'ok', data: data}); if (cb) { // JSONP response — bypasses browser CORS completely return ContentService.createTextOutput(cb + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } // ── POST: Write data (no-cors, fire-and-forget) ── function doPost(e) { try { var d = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(d.tab) || ss.insertSheet(d.tab); if (d.action === 'sync') { sh.clearContents(); if (d.rows && d.rows.length > 0) sh.getRange(1,1,d.rows.length,d.rows[0].length).setValues(d.rows); } if (d.action === 'append') sh.appendRow(d.row); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
⚡ Push = WebApp → Sheet  |  Pull = Sheet → WebApp
